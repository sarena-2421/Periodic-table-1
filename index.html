// ================= 2. 元素單字卡測驗 (關卡一) =================

function ElementQuiz() {

  const [questions, setQuestions] = useState([]);

  const [currentQ, setCurrentQ] = useState(0);

  const [score, setScore] = useState(0);

  const [showResult, setShowResult] = useState(false);

  const [feedback, setFeedback] = useState(null);

 

  const initQuiz = () => {

    const newQs = [];

    for (let i = 0; i < 10; i++) {

      // 隨機選一個目標元素

      const target = allElements[Math.floor(Math.random() * allElements.length)];

      const askSymbol = Math.random() > 0.5; // true = 看中文猜英文符號, false = 看英文符號猜中文

     

      // 產生 4 個不重複的選項

      let options = new Set([askSymbol ? target.s : target.n]);

      while(options.size < 4) {

         const randomEl = allElements[Math.floor(Math.random() * allElements.length)];

         options.add(askSymbol ? randomEl.s : randomEl.n);

      }

      const optionsArr = Array.from(options).sort(() => 0.5 - Math.random());

     

      newQs.push({

        targetDisplay: askSymbol ? target.n : target.s,

        qText: askSymbol ? `請選出「${target.n}」的英文符號` : `請問「${target.s}」是哪一個中文元素？`,

        options: optionsArr,

        ans: optionsArr.indexOf(askSymbol ? target.s : target.n),

        hint: `${target.s} = ${target.n} (${target.b})`

      });

    }

    setQuestions(newQs);

    setCurrentQ(0);

    setScore(0);

    setShowResult(false);

    setFeedback(null);

  };

 

  useEffect(() => {

    initQuiz();

  }, []);

 

  const handleAnswer = (index) => {

    if (feedback) return; // 防止重複點擊

 

    const isCorrect = index === questions[currentQ].ans;

    if (isCorrect) setScore(score + 1);

 

    setFeedback({

      correct: isCorrect,

      text: isCorrect ? '✅ 答對了！' : `❌ 答錯了。正確答案是 ${questions[currentQ].options[questions[currentQ].ans]}。`,

      hint: questions[currentQ].hint

    });

 

    setTimeout(() => {

      setFeedback(null);

      if (currentQ + 1 < questions.length) {

        setCurrentQ(currentQ + 1);

      } else {

        setShowResult(true);

      }

    }, 2000);

  };

 

  if (questions.length === 0) return <div>載入中...</div>;

 

  if (showResult) {

    return (

      <div className="text-center py-10 animate-fade-in">

        <h2 className="text-3xl font-bold mb-4">單字卡測驗結束！</h2>

        <p className="text-2xl mb-8">你的得分：<span className="text-amber-600 font-black text-4xl">{score}</span> / {questions.length}</p>

        <button onClick={initQuiz} className="bg-amber-500 text-white px-8 py-4 rounded-xl font-bold text-xl hover:bg-amber-600 shadow-md transition transform hover:scale-105">

          🔄 隨機換下一批單字卡

        </button>

      </div>

    );

  }

 

  const q = questions[currentQ];

 

  return (

    <div className="max-w-2xl mx-auto py-4 animate-fade-in">

      <div className="flex justify-between items-center mb-4 bg-gray-100 p-4 rounded-xl">

        <div className="text-sm text-gray-500 font-bold">進度：{currentQ + 1} / {questions.length} 題</div>

        <div className="text-sm text-amber-600 font-bold">目前得分：{score}</div>

      </div>

     

      {/* 視覺單字卡區塊 */}

      <div className="bg-amber-50 border-4 border-amber-300 p-12 rounded-3xl my-6 shadow-md text-center">

        <div className="text-lg font-bold text-amber-600 mb-4">{q.qText}</div>

        <h2 className="text-7xl font-black text-gray-800 tracking-wider drop-shadow-sm">{q.targetDisplay}</h2>

      </div>

     

      <div className="grid grid-cols-2 gap-4 mb-6">

        {q.options.map((opt, idx) => (

          <button

            key={idx}

            onClick={() => handleAnswer(idx)}

            disabled={!!feedback}

            className={`p-6 text-3xl font-bold rounded-2xl border-2 transition-all ${

              feedback && idx === q.ans ? 'bg-green-200 border-green-500 shadow-lg transform scale-105' :

              feedback && idx !== q.ans ? 'bg-gray-100 border-gray-200 text-gray-400 opacity-50' :

              'bg-white border-amber-200 text-gray-700 hover:bg-amber-50 hover:border-amber-400 hover:shadow-md'

            }`}

          >

            {opt}

          </button>

        ))}

      </div>

 

      {feedback && (

        <div className={`p-4 rounded-xl border-2 animate-bounce shadow-sm ${feedback.correct ? 'bg-green-100 border-green-400 text-green-800' : 'bg-red-100 border-red-400 text-red-800'}`}>

          <div className="font-bold text-xl mb-1">{feedback.text}</div>

          <div className="text-md mt-1">💡 記憶提示：{feedback.hint}</div>

        </div>

      )}

    </div>

  );

// ================= 3. 化合物配對 (價數和為零) =================

function CompoundGame() {

  const [cat, setCat] = useState(cations[0]);

  const [an, setAn] = useState(anions[0]);

  const [inputCat, setInputCat] = useState('');

  const [inputAn, setInputAn] = useState('');

  const [message, setMessage] = useState(null);

 

  const newQuestion = () => {

    setCat(cations[Math.floor(Math.random() * cations.length)]);

    setAn(anions[Math.floor(Math.random() * anions.length)]);

    setInputCat('');

    setInputAn('');

    setMessage(null);

  };

 

  useEffect(() => {

    newQuestion();

  }, []);

 

  const checkAnswer = () => {

    const x = parseInt(inputCat);

    const y = parseInt(inputAn);

   

    if (isNaN(x) || isNaN(y) || x <= 0 || y <= 0) {

      setMessage({ type: 'error', text: '請輸入大於 0 的正整數喔！' });

      return;

    }

 

    // Check sum to zero

    const totalCharge = (x * cat.charge) + (y * an.charge);

   

    // Check if it's the simplest ratio

    const d = gcd(x, y);

    const isSimplest = d === 1;

 

    if (totalCharge === 0) {

      if (isSimplest) {

        setMessage({

          type: 'success',

          text: `🎉 完美！ ${x}×(+${cat.charge}) + ${y}×(${an.charge}) = 0。總電荷為零！`

        });

      } else {

        setMessage({

          type: 'warning',

          text: `🤔 電荷平衡了，但不是最簡整數比喔！試試約分除以 ${d}。`

        });

      }

    } else {

      setMessage({

        type: 'error',

        text: `💥 不平衡喔！目前總電荷為 ${totalCharge > 0 ? '+' : ''}${totalCharge}。需要讓正負抵銷為 0。`

      });

    }

  };

 

  return (

    <div className="py-6 animate-fade-in text-center">

      <div className="mb-8">

        <h2 className="text-2xl font-bold text-indigo-800 mb-2">⚖️ 組合化合物 (價數和為零)</h2>

        <p className="text-gray-600">你需要幾個陽離子和幾個陰離子，才能讓化合物的「總電荷」為 0 呢？(請寫最簡整數比)</p>

      </div>

 

      <div className="flex flex-col md:flex-row justify-center items-center gap-6 mb-8">

        {/* 陽離子 */}

        <div className="bg-red-50 border-4 border-red-300 p-6 rounded-2xl w-48 text-center shadow-lg">

          <div className="text-4xl font-bold text-red-600 mb-2">{renderIon(cat)}</div>

          <div className="text-sm font-bold text-red-800">{cat.name}</div>

          <div className="mt-4">

            <input

              type="number"

              min="1"

              value={inputCat}

              onChange={(e) => setInputCat(e.target.value)}

              placeholder="數量"

              className="w-24 p-2 text-center text-xl font-bold rounded border-2 border-red-300 focus:outline-none focus:border-red-500"

            />

            <span className="ml-2 font-bold text-gray-500">個</span>

          </div>

        </div>

 

        <div className="text-5xl font-black text-gray-300">+</div>

 

        {/* 陰離子 */}

        <div className="bg-blue-50 border-4 border-blue-300 p-6 rounded-2xl w-48 text-center shadow-lg">

          <div className="text-4xl font-bold text-blue-600 mb-2">{renderIon(an)}</div>

          <div className="text-sm font-bold text-blue-800">{an.name}</div>

          <div className="mt-4">

            <input

              type="number"

              min="1"

              value={inputAn}

              onChange={(e) => setInputAn(e.target.value)}

              placeholder="數量"

              className="w-24 p-2 text-center text-xl font-bold rounded border-2 border-blue-300 focus:outline-none focus:border-blue-500"

function IonSortGame() {

  const allIons = [...cations, ...anions];

  const [currentIon, setCurrentIon] = useState(allIons[0]);

  const [score, setScore] = useState(0);

  const [streak, setStreak] = useState(0);

  const [feedback, setFeedback] = useState(null);

 

  const nextIon = () => {

    const randomIon = allIons[Math.floor(Math.random() * allIons.length)];

    setCurrentIon(randomIon);

    setFeedback(null);

  };

 

  useEffect(() => {

    nextIon();

  }, []);

 

  const handleSort = (type) => {

    if (feedback) return;

 

    const isCation = currentIon.charge > 0;

    const correct = (type === 'cation' && isCation) || (type === 'anion' && !isCation);

 

    if (correct) {

      setScore(score + 10);

      setStreak(streak + 1);

      setFeedback({ correct: true, text: `✅ 正確！${currentIon.name} 是${isCation ? '陽' : '陰'}離子。` });

      setStreak(0);

    }

 

    setTimeout(nextIon, 1500);

  };

 

  return (

    <div className="py-6 text-center max-w-lg mx-auto">

      <div className="flex justify-between items-center mb-6 bg-gray-100 p-4 rounded-xl font-bold">

        <div className="text-xl text-indigo-700">分數: {score}</div>

    // 隨機打亂題庫並抽出指定數量的題目

    setQuizQuestions(shuffled.slice(0, count));

    setScore(0);

    setFeedback(null);

  };

        setGameState('result');

      }

      <div className="text-center py-10 animate-fade-in max-w-2xl mx-auto">

        <h2 className="text-3xl font-bold mb-4">測驗結束！</h2>

        <div className="text-6xl mb-6">

          {percentage >= 80 ? '🏆' : percentage >= 60 ? '👍' : '💪'}

        </div>

        <p className="text-2xl mb-4 text-gray-700">

          你的得分：<span className="text-indigo-600 font-black text-4xl">{score}</span> / {quizQuestions.length}

        </p>

        <p className="text-xl mb-8 font-bold text-gray-500">正確率：{percentage}%</p>

        <div className="flex justify-center gap-4">

          <button onClick={() => setGameState('setup')} className="bg-indigo-600 text-white px-8 py-3 rounded-xl font-bold hover:bg-indigo-700 transition">

            🔄 重新選擇題數

          </button>

        </div>

      </div>

    );

  }

 

  const q = quizQuestions[currentQ];

 

  return (

    <div className="max-w-3xl mx-auto py-4 animate-fade-in">

      <div className="flex justify-between items-center mb-6 bg-gray-100 p-4 rounded-xl">

        <div className="text-lg text-gray-500 font-bold">進度：{currentQ + 1} / {quizQuestions.length}</div>

        <div className="text-lg text-indigo-600 font-bold">目前得分：{score}</div>

      </div>

     

      <h2 className="text-2xl font-bold mb-6 text-gray-800 leading-relaxed">{q.q}</h2>

     

      <div className="grid grid-cols-1 md:grid-cols-2 gap-4 mb-6">

        {q.options.map((opt, idx) => (

          <button

            key={idx}

            onClick={() => handleAnswer(idx)}

            disabled={!!feedback}

            className={`p-5 text-lg font-bold rounded-xl border-2 transition-all ${

              feedback && idx === q.ans ? 'bg-green-200 border-green-500 text-green-900 shadow-md transform scale-105' :

              feedback && idx !== q.ans ? 'bg-gray-100 border-gray-200 text-gray-400 opacity-60' :

              'bg-white border-indigo-200 text-gray-700 hover:bg-indigo-50 hover:border-indigo-400 hover:shadow'

            }`}

          >

            {opt}

          </button>

        ))}

      </div>

 

      {feedback && (

        <div className={`p-4 rounded-xl border-2 animate-bounce shadow-sm ${feedback.correct ? 'bg-green-100 border-green-400 text-green-800' : 'bg-red-100 border-red-400 text-red-800'}`}>

          <div className="font-bold text-xl mb-1">{feedback.text}</div>

          <div className="text-md mt-2">💡 解析：{feedback.hint}</div>

        </div>

      )}

    </div>

  );

}
