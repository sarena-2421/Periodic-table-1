    function ElementQuiz() {

      const [questions, setQuestions] = useState([]);

        const newQs = [];

        for (let i = 0; i < 10; i++) {

          const target = allElements[Math.floor(Math.random() * allElements.length)];

          const askSymbol = Math.random() > 0.5;

          let options = new Set([askSymbol ? target.s : target.n]);

          while(options.size < 4) {

             const randomEl = allElements[Math.floor(Math.random() * allElements.length)];

             options.add(askSymbol ? randomEl.s : randomEl.n);

          }

          const optionsArr = Array.from(options).sort(() => 0.5 - Math.random());

          newQs.push({

            targetDisplay: askSymbol ? target.n : target.s,

            qText: askSymbol ? `請選出「${target.n}」的英文符號` : `請問「${target.s}」是哪一個中文元素？`,

            options: optionsArr,

            ans: optionsArr.indexOf(askSymbol ? target.s : target.n),

            hint: `${target.s} = ${target.n} (${target.b})`

          });

        }

        setQuestions(newQs);

        setCurrentQ(0);

        setScore(0);

        setShowResult(false);

      };

    // ================= 3. 化合物配對 (關卡二) =================

    function CompoundGame() {

      const [cat, setCat] = useState(cations[0]);

      const [an, setAn] = useState(anions[0]);

      const [inputCat, setInputCat] = useState('');

      const [inputAn, setInputAn] = useState('');

      const [message, setMessage] = useState(null);

 

      const newQuestion = () => {

        setCat(cations[Math.floor(Math.random() * cations.length)]);

        setAn(anions[Math.floor(Math.random() * anions.length)]);

        setInputCat('');

        setInputAn('');

        setMessage(null);

      };

 

      useEffect(() => { newQuestion(); }, []);

 

      const checkAnswer = () => {

        const x = parseInt(inputCat);

        const y = parseInt(inputAn);

        if (isNaN(x) || isNaN(y) || x <= 0 || y <= 0) {

          setMessage({ type: 'error', text: '請輸入大於 0 的正整數喔！' }); return;

        }

        const totalCharge = (x * cat.charge) + (y * an.charge);

        const d = gcd(x, y);

        if (totalCharge === 0) {

          if (d === 1) setMessage({ type: 'success', text: `🎉 完美！ ${x}×(+${cat.charge}) + ${y}×(${an.charge}) = 0。` });

          else setMessage({ type: 'warning', text: `🤔 電荷平衡了，但不是最簡整數比喔！試試約分除以 ${d}。` });

        } else {

          setMessage({ type: 'error', text: `💥 不平衡喔！目前總電荷為 ${totalCharge > 0 ? '+' : ''}${totalCharge}。` });

        }

      };

 

      return (

        <div className="py-6 animate-fade-in text-center">

          <div className="mb-8">

            <h2 className="text-2xl font-bold text-indigo-800 mb-2">⚖️ 組合化合物 (價數和為零)</h2>

            <p className="text-gray-600">你需要幾個陽離子和幾個陰離子，才能讓化合物的「總電荷」為 0 呢？(請寫最簡整數比)</p>

          </div>

          <div className="flex flex-col md:flex-row justify-center items-center gap-6 mb-8">

            <div className="bg-red-50 border-4 border-red-300 p-6 rounded-2xl w-48 text-center shadow-lg">

              <div className="text-4xl font-bold text-red-600 mb-2">{renderIon(cat)}</div>

              <div className="text-sm font-bold text-red-800">{cat.name}</div>

              <div className="mt-4">

                <input type="number" min="1" value={inputCat} onChange={(e) => setInputCat(e.target.value)} placeholder="數量" className="w-24 p-2 text-center text-xl font-bold rounded border-2 border-red-300" />

                <span className="ml-2 font-bold text-gray-500">個</span>

              </div>

            </div>

            <div className="text-5xl font-black text-gray-300">+</div>

            <div className="bg-blue-50 border-4 border-blue-300 p-6 rounded-2xl w-48 text-center shadow-lg">

              <div className="text-4xl font-bold text-blue-600 mb-2">{renderIon(an)}</div>

              <div className="text-sm font-bold text-blue-800">{an.name}</div>

              <div className="mt-4">

                <input type="number" min="1" value={inputAn} onChange={(e) => setInputAn(e.target.value)} placeholder="數量" className="w-24 p-2 text-center text-xl font-bold rounded border-2 border-blue-300" />

                <span className="ml-2 font-bold text-gray-500">個</span>

              </div>

            </div>

          </div>

          <div className="space-x-4 mb-6">

            <button onClick={checkAnswer} className="bg-green-500 text-white px-8 py-3 rounded-xl font-bold text-xl hover:bg-green-600 shadow transition">送出檢查 ✅</button>

            {message && message.type === 'success' && <button onClick={newQuestion} className="bg-indigo-600 text-white px-8 py-3 rounded-xl font-bold text-xl hover:bg-indigo-700 shadow transition">下一題 ➡️</button>}

            </div>

          )}

        </div>

      );

    }

 

    // ================= 4. 陰陽離子分類賽 (關卡二) =================

    function IonSortGame() {

      const allIons = [...cations, ...anions];

      const [currentIon, setCurrentIon] = useState(allIons[0]);

      const [score, setScore] = useState(0);

      const [streak, setStreak] = useState(0);

      const [feedback, setFeedback] = useState(null);

 

      const nextIon = () => { setCurrentIon(allIons[Math.floor(Math.random() * allIons.length)]); setFeedback(null); };

      useEffect(() => { nextIon(); }, []);

 

      const handleSort = (type) => {

        if (feedback) return;

        const isCation = currentIon.charge > 0;

        const correct = (type === 'cation' && isCation) || (type === 'anion' && !isCation);

        if (correct) {

          setScore(score + 10); setStreak(streak + 1);

          setFeedback({ correct: true, text: `✅ 正確！${currentIon.name} 是${isCation ? '陽' : '陰'}離子。` });

        } else {

          setStreak(0);

          setFeedback({ correct: false, text: `❌ 錯誤！帶有 ${currentIon.chargeStr} 電荷的是${isCation ? '陽' : '陰'}離子。` });

        }

        setTimeout(nextIon, 1500);

      };

 

      return (

        <div className="py-6 text-center max-w-lg mx-auto animate-fade-in">

          <div className="flex justify-between items-center mb-6 bg-gray-100 p-4 rounded-xl font-bold">

            <div className="text-xl text-indigo-700">分數: {score}</div>

            <div className="text-xl text-orange-500">🔥 連勝: {streak}</div>

          </div>

          <div className="bg-white border-4 border-dashed border-indigo-200 p-10 rounded-3xl my-8 shadow-sm">

            <div className="text-6xl font-black text-gray-800 mb-2">{renderIon(currentIon)}</div>

            <div className="text-xl text-gray-600 font-bold">{currentIon.name}</div>

          </div>

          <div className="grid grid-cols-2 gap-6">

            <button onClick={() => handleSort('cation')} className="bg-red-100 hover:bg-red-200 border-4 border-red-400 text-red-800 font-bold text-2xl py-6 rounded-2xl transition transform hover:scale-105">陽離子<br/><span className="text-sm">(帶正電 +)</span></button>

            <button onClick={() => handleSort('anion')} className="bg-blue-100 hover:bg-blue-200 border-4 border-blue-400 text-blue-800 font-bold text-2xl py-6 rounded-2xl transition transform hover:scale-105">陰離子<br/><span className="text-sm">(帶負電 -)</span></button>

          </div>

          {feedback && <div className={`mt-6 p-4 rounded-xl font-bold text-lg ${feedback.correct ? 'bg-green-100 text-green-700' : 'bg-red-100 text-red-700'}`}>{feedback.text}</div>}

        </div>

      );

    }
